\appendix
\section{Implementation Pseudocode}

\subsection{Target Extraction}

\begin{verbatim}
for each (x, y) grid sample:
    hits = sort(vertical_ray_intersections(mesh, x, y))
    top_z[x, y] = highest(hits)
    terrace_z[x, y] = highest(hit below top_z by at least gap)
filter top and terrace maps by slope + morphology
couple low anchor to high anchor outside terrace mask
\end{verbatim}

\subsection{Anchor Solve}

\begin{verbatim}
build LS system:
    boundary consistency terms
    gradient shaping terms
    weak gauge regularization
solve with sparse least squares
project with Lipschitz / conical clamp
\end{verbatim}

\subsection{Rank-2 Construction}

\begin{verbatim}
S_high = solve_anchor(top targets)
S_low  = solve_anchor(terrace targets)
for each query height z_s:
    b = smoothstep((z_s - z0) / (z1 - z0))
    D = (1 - b) * S_low + b * S_high
verify 1 + dD/dz_s > 0
\end{verbatim}
